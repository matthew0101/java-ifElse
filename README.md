# java-ifElse
basic code for if/else if/if

public class IfElse

    public static void main(String[]args){
        int x=100;

        if (x>=100){
            System.out.println("RICH");

        }else if (x>=70){
            System.out.println("COMFORT");

        }else if (x>=50){
            System.out.println("NORMAL");

        }else if (x>=20){
            System.out.println("SMALL");

        }else 
        System.out.println("POOR");
          }
}
